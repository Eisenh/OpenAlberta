# i18n System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add multilingual support — English strings ship bundled; other languages are generated once by Claude via a `translate-ui` edge function, cached in the DB and localStorage. Language preference persists via cookie and user metadata. Stale translations auto-regenerate when UI strings change (detected via build-time hash).

**Architecture:** `src/lib/i18n/en.ts` is the single source of truth. A Vite plugin hashes it at build time and injects `VITE_STRINGS_HASH`. A Svelte store (`src/lib/i18n/store.ts`) loads translations on language switch. The `translate-ui` edge function calls Claude and writes to the `translations` table. Admin page shows translation health.

**Tech Stack:** Svelte 5, TypeScript, Vite plugin API, Supabase Edge Functions (Deno), Claude API (`claude-opus-4-6`)

**Prerequisite:** Plans 1–4 complete — schema includes `translations` table, `supabaseClient.ts` uses publishable key.

- [ ] **Step 2: Add the hash plugin to `vite.config.ts`**

  Add the plugin at the top of the file and reference it in the plugins array. Replace the existing `vite.config.ts` content with:

  ```typescript
  import { defineConfig, loadEnv, type Plugin } from 'vite'
  import { svelte } from '@sveltejs/vite-plugin-svelte'
  import { resolve } from 'path'
  import fs from 'fs'
  import crypto from 'crypto'

  function stringsHashPlugin(): Plugin {
    return {
      name: 'strings-hash',
      config() {
        const content = fs.readFileSync(resolve(__dirname, 'src/lib/i18n/en.ts'), 'utf-8')
        const hash = crypto.createHash('sha256').update(content).digest('hex').slice(0, 16)
        return {
          define: {
            'import.meta.env.VITE_STRINGS_HASH': JSON.stringify(hash),
          },
        }
      },
    }
  }

  export default defineConfig(({ mode }) => {
    const env = loadEnv(mode, process.cwd(), '')
    const isDev = mode === 'development'

    const serverConfig = isDev
      ? {
          https: {
            key: fs.readFileSync('localhost-key.pem'),
            cert: fs.readFileSync('localhost.pem'),
          },
          host: 'localhost',
          port: 5173,
          headers: { 'Access-Control-Allow-Origin': '*' },
          fs: { strict: false, allow: ['.'] },
        }
      : {}

    return {
      base: '/OpenAlberta/',
      publicDir: 'public',
      define: {
        'process.env': { ...env, GITHUB_PAGES: mode === 'production' },
      },
      plugins: [svelte(), stringsHashPlugin()],
      resolve: {
        alias: {
          $lib: resolve('./src/lib'),
          $stores: resolve('./src/lib/stores'),
        },
      },
      server: { ...serverConfig, open: true },
      build: {
        rollupOptions: {
          input: { main: resolve(__dirname, 'index.html') },
        },
      },
    }
  })
  ```

  ```typescript
  import { serve } from 'https://deno.land/std@0.177.0/http/server.ts'
  import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
  import Anthropic from 'https://esm.sh/@anthropic-ai/sdk@0.27.0'

  const corsHeaders = {
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
  }

  serve(async (req) => {
    if (req.method === 'OPTIONS') {
      return new Response('ok', { headers: corsHeaders })
    }

    // Require authentication — prevents abuse
    const authHeader = req.headers.get('Authorization')
    if (!authHeader) {
      return Response.json({ error: 'Unauthorized' }, { status: 401, headers: corsHeaders })
    }

    const supabaseAdmin = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
    )

    const { data: { user }, error: authError } = await supabaseAdmin.auth.getUser(
      authHeader.replace('Bearer ', '')
    )
    if (authError || !user) {
      return Response.json({ error: 'Unauthorized' }, { status: 401, headers: corsHeaders })
    }

    const { locale, strings, stringsHash } = await req.json() as {
      locale: string
      strings: Record<string, { text: string; context: string }>
      stringsHash: string
    }

    const anthropic = new Anthropic({ apiKey: Deno.env.get('ANTHROPIC_API_KEY')! })

    // Build the translation prompt
    const stringLines = Object.entries(strings)
      .map(([key, { text, context }]) => `${key}: ${JSON.stringify(text)}  // Context: ${context}`)
      .join('\n')

    const message = await anthropic.messages.create({
      model: 'claude-opus-4-6',
      max_tokens: 4096,
      system: `You are a professional translator for a government open data portal application. 
The app helps citizens discover and search datasets published by government agencies.
Translate all UI strings accurately and naturally for the target language.
Preserve technical terms like "CKAN", "dataset", and ISO codes exactly as-is.
Respond with ONLY a valid JSON object mapping each key to its translated string value, nothing else.`,
      messages: [
        {
          role: 'user',
          content: `Translate these UI strings to ${locale}. 
Each line is: key: "English text"  // Context: explanation
Return a JSON object: { "key": "translated text", ... }

${stringLines}`,
        },
      ],
    })

    let translations: Record<string, string>
    try {
      const content = message.content[0]
      if (content.type !== 'text') throw new Error('Unexpected response type')
      // Strip any markdown code fences if present
      const json = content.text.replace(/^```json?\n?/, '').replace(/\n?```$/, '').trim()
      translations = JSON.parse(json)
    } catch (e) {
      return Response.json({ error: `Failed to parse Claude response: ${e}` }, { status: 500, headers: corsHeaders })
    }

    // Store in translations table
    await supabaseAdmin.from('translations').upsert({
      locale,
      translations,
      strings_hash: stringsHash,
      generated_at: new Date().toISOString(),
      generated_by: 'claude-opus-4-6',
    }, { onConflict: 'locale' })

    return Response.json({ translations }, { headers: corsHeaders })
  })
  ```

## Task 4: Create the i18n store

**Files:**

- Create: `src/lib/i18n/store.ts`

- [ ] **Step 1: Create `src/lib/i18n/store.ts`**

  ```typescript
  import { writable, derived } from 'svelte/store'
  import { strings } from './en'
  import { supabase } from '../supabaseClient'

  const STRINGS_HASH = import.meta.env.VITE_STRINGS_HASH as string

  // Loaded translations per locale — keyed by locale code
  const loadedTranslations = writable<Record<string, Record<string, string>>>({ en: {} })

  export const locale = writable<string>('en')

  // t(key) returns translated string, falling back to English
  export const t = derived(
    [locale, loadedTranslations],
    ([$locale, $loaded]) =>
      (key: string): string =>
        $loaded[$locale]?.[key] ?? strings[key]?.text ?? key
  )

  /** Call this on app load to restore saved locale preference */
  export function initLocale(preferredLocale?: string | null): void {
    const cookieLocale = getCookie('locale')
    const browserLocale = navigator.language.split('-')[0]
    const chosen = preferredLocale || cookieLocale || browserLocale || 'en'
    if (chosen !== 'en') {
      switchLocale(chosen)
    }
  }

  /** Switch to a locale, loading translations if needed */
  export async function switchLocale(newLocale: string): Promise<void> {
    if (newLocale === 'en') {
      locale.set('en')
      setCookie('locale', 'en')
      return
    }

    // Check localStorage cache first
    const cacheKey = `translations.${newLocale}`
    const cached = localStorage.getItem(cacheKey)
    if (cached) {
      try {
        const { hash, data } = JSON.parse(cached) as { hash: string; data: Record<string, string> }
        if (hash === STRINGS_HASH) {
          applyTranslations(newLocale, data)
          return
        }
      } catch { /* cache corrupt — fall through */ }
    }

    // Check DB
    const { data: row } = await supabase
      .from('translations')
      .select('translations, strings_hash')
      .eq('locale', newLocale)
      .single()

    if (row && row.strings_hash === STRINGS_HASH) {
      applyTranslations(newLocale, row.translations)
      localStorage.setItem(cacheKey, JSON.stringify({ hash: STRINGS_HASH, data: row.translations }))
      return
    }

    // Generate via edge function (requires auth — falls back to English if not logged in)
    const { data: { session } } = await supabase.auth.getSession()
    if (!session) {
      // Not logged in — can't generate; apply English silently
      locale.set(newLocale) // still update locale for cookie/preference
      setCookie('locale', newLocale)
      return
    }

    const { data: fnData, error } = await supabase.functions.invoke('translate-ui', {
      body: { locale: newLocale, strings, stringsHash: STRINGS_HASH },
    })

    if (!error && fnData?.translations) {
      applyTranslations(newLocale, fnData.translations)
      localStorage.setItem(cacheKey, JSON.stringify({ hash: STRINGS_HASH, data: fnData.translations }))
    }
  }

  /** Save user's locale preference to their Supabase profile */
  export async function saveLocalePreference(localeCode: string): Promise<void> {
    const { data: { session } } = await supabase.auth.getSession()
    if (session) {
      await supabase.auth.updateUser({ data: { preferred_locale: localeCode } })
    }
    setCookie('locale', localeCode)
  }

  function applyTranslations(localeCode: string, data: Record<string, string>): void {
    loadedTranslations.update(all => ({ ...all, [localeCode]: data }))
    locale.set(localeCode)
    setCookie('locale', localeCode)
  }

  function getCookie(name: string): string | null {
    const match = document.cookie.match(new RegExp(`(?:^|; )${name}=([^;]*)`))
    return match ? decodeURIComponent(match[1]) : null
  }

  function setCookie(name: string, value: string): void {
    document.cookie = `${name}=${encodeURIComponent(value)}; SameSite=Lax; Max-Age=31536000; Path=/`
  }
  ```

  ```javascript
  import { strings } from '../i18n/en'
  import { supabase } from '../supabaseClient'

  const STRINGS_HASH = import.meta.env.VITE_STRINGS_HASH

  let translations: Array<{
    locale: string
    generated_at: string
    strings_hash: string
    generated_by: string
  }> = []
  let translationsLoading = true
  let regenerating: string | null = null

  async function loadTranslations() {
    translationsLoading = true
    const { data } = await supabase
      .from('translations')
      .select('locale, generated_at, strings_hash, generated_by')
      .order('locale')
    translations = data ?? []
    translationsLoading = false
  }

  async function regenerateLocale(localeCode: string) {
    regenerating = localeCode
    const { data: { session } } = await supabase.auth.getSession()
    await supabase.functions.invoke('translate-ui', {
      body: { locale: localeCode, strings, stringsHash: STRINGS_HASH },
    })
    await loadTranslations()
    // Bust localStorage cache for this locale
    localStorage.removeItem(`translations.${localeCode}`)
    regenerating = null
  }

  async function regenerateAll() {
    for (const t of translations) {
      await regenerateLocale(t.locale)
    }
  }

  // Load translations when admin panel mounts
  import { onMount } from 'svelte'
  // (add to existing onMount or create one)
  // Inside onMount: await loadTranslations()
  ```
